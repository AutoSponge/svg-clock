    The startProcess sends elements to be created then beings the animationFrameProcess

    startProcess

    • null ‘start’ -> IN
    • OUT -> null ‘init’
    • OUT -> bool ‘animate’
    module.exports = function startProcess( app ) {

    function startComponent() {

        app.emit( 'init' );

        app.emit( 'animate', true );

    }

    app.on( 'start', startComponent );

};
